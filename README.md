        "    return re.sub(r\"\\s+\", \" \", str(text)).strip()\n",

        "\n",
        "def strip_accents(text: str) -> str:\n",
        "    normalised = unicodedata.normalize(\"NFKD\", str(text))\n",
        "    return \"\".join(char for char in normalised if not unicodedata.combining(char))\n",
        "\n",
        "\n",
        "def normalise(text: str) -> str:\n",
        "    text = strip_accents(str(text).lower())\n",
